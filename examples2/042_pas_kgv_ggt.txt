Program Biorhythmus_analyser;

var temp: integer;

function ggt(m,n: integer): integer;
begin
  if (n=0) then
    result:= m
  else
    result:= ggt(n, m mod n);
end;

function kgv(m,n: integer): integer;
var o, p: integer;
begin
  o:= ggt(m,n);
  p:= ((m * n) div o);
  result:= p;
end;
